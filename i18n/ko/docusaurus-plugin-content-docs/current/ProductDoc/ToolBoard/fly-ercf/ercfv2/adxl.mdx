# 가속도계 사용


import ADXL from "/docs/DebugDoc/faq/kipper-adxl.mdx";

## Klipper 전환

    * 아직 병합되지 않았으므로 Klipper를 전환해야 사용할 수 있습니다

    ```bash
    sudo cp -r klipper klipperdev
    sudo rm -rf ~/klipper
    git clone https://github.com/FLYmaker/klipper.git
    sudo systemctl restart klipper
    ```

    * 참고 설정
    ```cfg
    #####################################################################
    #                       LIS3DH 가속도계 설정（필요 시 활성화）
    #####################################################################
    [lis2dw]
    is_lis3dh:True
    cs_pin: SHT36:gpio9
    spi_software_sclk_pin: SHT36:gpio10
    spi_software_mosi_pin: SHT36:gpio11
    spi_software_miso_pin: SHT36:gpio12

    [resonance_tester]
    accel_chip: lis2dw
    probe_points:
        125, 125, 20  #여기에서 공진 측정 시 노즐의 좌표를 설정합니다


```
