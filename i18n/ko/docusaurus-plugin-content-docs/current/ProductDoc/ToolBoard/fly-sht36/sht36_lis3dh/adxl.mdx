# 가속도계 사용

import ADXL from "/docs/DebugDoc/faq/kipper-adxl.mdx";

## Klipper 업데이트

    * Klipper는 2024년 11월 13일 LIS3DH 가속도계를 지원합니다. LIS3DH 가속도계를 사용하려면 Klipper를 업데이트해야 합니다.
    * 네트워크로 상위 컴퓨터에 연결한 후 아래 코드를 입력하십시오(네트워크 문제로 인해 업데이트가 실패할 수 있음):
        ```bash
        cd ~/klipper 
        git pull
        ```
## 설정 추가

    ```cfg
    #####################################################################
    #                       LIS3DH 가속도계 설정(필요 시 활성화)
    #####################################################################
    [lis3dh]
    cs_pin: SHT36:gpio9
    spi_software_sclk_pin: SHT36:gpio10
    spi_software_mosi_pin: SHT36:gpio11
    spi_software_miso_pin: SHT36:gpio12

    [resonance_tester]
    accel_chip: lis3dh
    probe_points:
        125, 125, 20  # 여기서 공진 측정 시 노즐의 위치를 설정하십시오


```
