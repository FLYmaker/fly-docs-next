# Подключение материнской платы

    * В этом руководстве используется пример PI V2
        <ImageView image={require('@site/docs/ProductDoc/SBC/fly-pi-v2/img/adxl345.webp').default} size="80%" align="left" />

## Настройка

    ```cfg
    [mcu host]
    serial: /tmp/klipper_host_mcu

    [adxl345]
    cs_pin: host:None
    spi_bus: spidev1.0

    [resonance_tester]
    accel_chip: adxl345
    #accel_chip_y: adxl345 bed
    probe_points:
        100, 100, 20  # пример

    ```
