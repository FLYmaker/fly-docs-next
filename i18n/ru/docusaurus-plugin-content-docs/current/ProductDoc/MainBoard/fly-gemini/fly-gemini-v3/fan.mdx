# Встроенный разъем для 5В вентилятора

## На плате Gemini есть встроенный 5В управляемый интерфейс для охлаждения основной платы, способ подключения показан ниже:

    <ImageView image={require('@site/docs/ProductDoc/MainBoard/fly-gemini/fly-gemini-v3/img/corefan.png').default} size="80%" align="center" />

## Конфигурация для справки

    ```cfg
    [mcu host]                      # FlyOS уже создал для вас MCU Linux, просто определите его для использования            
    serial: /tmp/klipper_host_mcu 

    [temperature_sensor FLY-π]      # Температура верхнего уровня
    sensor_type: temperature_host


    [temperature_fan core_fan]      # Вентилятор охлаждения верхнего уровня
    pin: host:gpiochip1/gpio103     # Пин вентилятора верхнего уровня
    max_power: 1.0
    sensor_type: temperature_host   # Установить как температуру основного контроллера верхнего уровня
    control:watermark               # Способ управления
    target_temp: 48                 # Температура включения вентилятора верхнего уровня
    min_temp: 0                     # Минимальная температура, ниже которой будет ошибка
    max_temp: 90                    # Максимальная температура, выше которой будет ошибка
    off_below: 0.10
    kick_start_time: 0.50
    max_speed: 0.8                  # Максимальная скорость, 80% от полной мощности
    min_speed: 0.3                  # Минимальная скорость, 30% от полной мощности
    ```
