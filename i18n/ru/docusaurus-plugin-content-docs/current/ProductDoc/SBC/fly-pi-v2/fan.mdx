# Встроенный разъем для вентилятора 5V

## Плата Pi-V2 оснащена контролируемым интерфейсом 5V для охлаждения основного контроллера, способ подключения показан на следующем изображении
    <ImageView image={require('@site/docs/ProductDoc/SBC/fly-pi-v2/img/fan.png').default} size="80%" align="left" />


## Пример конфигурации

    ```cfg
    [mcu host]                      # FlyOS уже сгенерировал для вас MCU Linux, просто определите и используйте
    serial: /tmp/klipper_host_mcu 

    [temperature_sensor FLY-π]      # Температура хоста
    sensor_type: temperature_host


    [temperature_fan core_fan]      # Вентилятор охлаждения хоста
    pin: host:gpiochip1/gpio103     # Пин вентилятора хоста
    max_power: 1.0
    sensor_type: temperature_host   # Установите на температуру хоста
    control:watermark               # Способ управления
    target_temp: 48                 # Температура запуска вентилятора хоста
    min_temp: 0                     # Минимальная температура, ниже которой будет выдаваться ошибка
    max_temp: 90                    # Максимальная температура, выше которой будет выдаваться ошибка
    off_below: 0.10
    kick_start_time: 0.50
    max_speed: 0.8                  # Максимальная скорость, 80% от полной мощности
    min_speed: 0.3                  # Минимальная скорость, 30% от полной мощности
    ```
